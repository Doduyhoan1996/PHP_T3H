từ khoá IN cho phép bạn lọc nhiều giá trị trong 1 cột với mệnh đề WHERE
    từ khoá IN là viết tắt của mệnh đề WHERE OR
    cú pháp :
    SELECT column_name(s)
    FROM table_name
    WHERE column_name IN (value1, value2, ...);

    ví dụ :
    SELECT * FROM customers WHERE id IN(1,2,3)
    lất ra các bản ghi có id là 1 hay 2 hay 3
