Câu lệnh ORDER BY ( sắp xếp )
    ASC tăng dần
    DESC giảm dần
    ví dụ 1:
    sắp xếp khách hàng theo  thứ tự id tăng dần
    SELECT * FROM customers ORDER BY id ASC
    ví dụ 2:
    sắp xếp khách hàng theo  thứ tự id giảm dần
    SELECT * FROM customers ORDER BY id DESC
    ví dụ 3:
    sắp xếp nhiều cột
    sắp xếp các khách hàng theo tên thành phố tăng dần A - Z
    và sắp xếp chức vụ giảm dần Z - A
    SELECT id, city , job_title FROM customers ORDER BY city ASC, job_title DESC
